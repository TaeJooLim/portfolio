	// 외근/출장 신청서
	@RequestMapping("admin/business_trip_form")
	public ModelAndView getBusinessTripForm() {
		ModelAndView mav = new ModelAndView();
		/* 현재날짜 */
		Date now = new Date();
		SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
		
		mav.addObject("now", sdf.format(now));
		mav.setViewName("/erp/admin/erp_attend/business_trip_form");
		
		return mav;
	}
