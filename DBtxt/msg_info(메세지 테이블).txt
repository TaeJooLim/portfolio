create table msg_info (
	msg_id int not null auto_increment primary key,	// PK
	msg_send_id_fk int not null, 			// 받는사람 (emp_id 저장)
	msg_from_id_fk int not null, 			// 보낸사람 (emp_id 저장)
	msg_subject varchar(50),			// 제목
	msg_content varchar(300),			// 내용
	msg_regdate datetime,				// 보낸시간
	msg_confirm char(1) default 'N', 		// 읽음 유/무 확인
	msg_send_del char(1) default 'N', 		// 보낸메세지 표시 유/무 (삭제기능) 
	msg_from_del char(1) default 'N' 		// 받은메세지 표시 유/무 (삭제기능)
);


create table msg_info (
	msg_id int not null auto_increment primary key,
	msg_send_id_fk int not null, 
	msg_from_id_fk int not null, 
	msg_subject varchar(50), 
	msg_content varchar(300),
	msg_regdate datetime, 
	msg_confirm char(1) default 'N', 
	msg_send_del char(1) default 'N', 
	msg_from_del char(1) default 'N' 
);
