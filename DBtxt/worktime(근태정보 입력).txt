create table worktime(
	wt_id int not null auto_increment primary key, //기본키
	wt_workdate char(10), // 근무일
	wt_starttime date, // 출근시간
	wt_endtime date, // 퇴근시간
	wt_workplace char(1), // 내외근 - selectbox(Y/N - 내근Y)
	wt_late char(1) // 지각 - Y/N
	
	emp_id_fk int not null,  // employee 테이블 조인
);

create table worktime(
	wt_id int not null auto_increment,
	wt_workdate date,
	wt_empNum varchar(20) not null,
	wt_department char(3),
	wt_starttime datetime,
	wt_endtime datetime,
	wt_workplace char(1),
	wt_late char(1),
	primary key(wt_id)
);
